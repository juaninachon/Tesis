% Level 1: Centered, Bold, Title Case
\titleformat{\section}
  {\normalfont\bfseries\centering}
  {}{0pt}{}

% Level 2: Flush Left, Bold, Title Case
\titleformat{\subsection}
  {\normalfont\bfseries}
  {}{0pt}{}

% Level 3: Flush Left, Bold Italic, Title Case
\titleformat{\subsubsection}
  {\normalfont\bfseries\itshape}
  {}{0pt}{}

% Level 4: Indented, Bold, Title Case, Ending with a Period. 
\titleformat{\paragraph}[runin]
  {\normalfont\bfseries}
  {}{0pt}{}[.] 

% Level 5: Indented, Bold Italic, Title Case, Ending with a Period. 
\titleformat{\subparagraph}[runin]
  {\normalfont\bfseries\itshape}
  {}{0pt}{}[.] 

% Make sure paragraphs after run-in headings start on same line
\titlespacing*{\paragraph}{\parindent}{1ex}{1em}
\titlespacing*{\subparagraph}{\parindent}{1ex}{1em}
